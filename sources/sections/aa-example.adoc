== Complete EXPRESS-Q file example

This annex provides a complete example of an EXPRESS-Q file for the Activity
module, demonstrating EXPRESS links for cross-schema references.

[source]
----
-- EXPRESS-Q file for the Activity module (ISO 10303-1047)

ENTITY_MAPPING <<express:Activity_arm.Activity,Activity>>;
EXTENSIBLE: FALSE;
AIM_ELEMENT: "<<express:action_schema.executed_action,executed_action>>";
EXPRESS_REF: ["action_schema", "basic_attribute_schema"];
REFPATH: {
  executed_action <= action
};

ATTRIBUTE_MAPPING id;
ASSERTION_TO: "id";
AIM_ELEMENT: "id_attribute.attribute_value";
REFPATH: {
  executed_action
  executed_action <= action
  action <- id_attribute.identified_item
  id_attribute.attribute_value
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING name;
ASSERTION_TO: "name";
AIM_ELEMENT: "action.name";
REFPATH: {
  executed_action
  executed_action <= action
  action.name
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING description;
ASSERTION_TO: "description";
AIM_ELEMENT: "action.description";
REFPATH: {
  executed_action
  executed_action <= action
  action.description
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING chosen_method;
ASSERTION_TO: "<<express:Activity_method_arm.Activity_method,Activity_method>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  executed_action
  executed_action <= action
  action.chosen_method -> action_method
  action_method
};
END_ATTRIBUTE_MAPPING;

END_ENTITY_MAPPING;

ENTITY_MAPPING <<express:Activity_arm.Activity_relationship,Activity_relationship>>;
EXTENSIBLE: FALSE;
AIM_ELEMENT: "<<express:action_schema.action_relationship,action_relationship>>";

ATTRIBUTE_MAPPING name;
ASSERTION_TO: "name";
AIM_ELEMENT: "action_relationship.name";
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING description;
ASSERTION_TO: "description";
AIM_ELEMENT: "action_relationship.description";
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING related_activity;
ASSERTION_TO: "<<express:Activity_arm.Activity,Activity>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  action_relationship
  action_relationship.related_action -> action
  action => executed_action
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING relating_activity;
ASSERTION_TO: "<<express:Activity_arm.Activity,Activity>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  action_relationship
  action_relationship.relating_action -> action
  action => executed_action
};
END_ATTRIBUTE_MAPPING;

END_ENTITY_MAPPING;

ENTITY_MAPPING <<express:Activity_arm.Activity_status,Activity_status>>;
EXTENSIBLE: FALSE;
AIM_ELEMENT: "<<express:action_schema.action_status,action_status>>";

ATTRIBUTE_MAPPING assigned_activity;
ASSERTION_TO: "<<express:Activity_arm.Activity,Activity>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  action_status
  action_status.assigned_action -> executed_action
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING status;
ASSERTION_TO: "status";
AIM_ELEMENT: "action_status.status";
END_ATTRIBUTE_MAPPING;

END_ENTITY_MAPPING;

ENTITY_MAPPING <<express:Activity_arm.Applied_activity_assignment,Applied_activity_assignment>>;
EXTENSIBLE: FALSE;
AIM_ELEMENT: "<<express:Activity_mim.applied_action_assignment,applied_action_assignment>>";
EXPRESS_REF: ["management_resources_schema"];

ATTRIBUTE_MAPPING assigned_activity;
ASSERTION_TO: "<<express:Activity_arm.Activity,Activity>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  applied_action_assignment
  applied_action_assignment <= action_assignment
  action_assignment
  action_assignment.assigned_action -> action
  action => executed_action
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING items;
ASSERTION_TO: "<<express:Activity_arm.activity_item,activity_item>>";
AIM_ELEMENT: "PATH";
REFPATH: {
  applied_action_assignment
  applied_action_assignment.items[i] -> action_items
};
END_ATTRIBUTE_MAPPING;

ATTRIBUTE_MAPPING role;
ASSERTION_TO: "role";
AIM_ELEMENT: "object_role.name";
REFPATH: {
  applied_action_assignment
  applied_action_assignment <= action_assignment
  action_assignment.role -> object_role
  object_role.name
};
END_ATTRIBUTE_MAPPING;

END_ENTITY_MAPPING;
----

This example demonstrates the complete mapping for the Activity module,
including:

- EXPRESS links for entity names (`<<express:Activity_arm.Activity,Activity>>`)
- EXPRESS links for AIM elements (`<<express:action_schema.executed_action,executed_action>>`)
- EXPRESS links for cross-module assertions (`<<express:Activity_method_arm.Activity_method,Activity_method>>`)
- Plain string AIM elements for attribute paths (`action.name`) and keywords (`PATH`)
- Reference paths with subtype, supertype, and attribute relationships
